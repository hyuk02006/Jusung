# (3) 날짜 정보에서 dt 접근자를 이용하여 연도(year)와 분기(quarter)를 구한다
def div_quar(cdate):
    if (cdate.month >=1) & (cdate.month <= 3):
        return 1
    elif (cdate.month >=4) & (cdate.month <= 6):
        return 2
    elif (cdate.month >=7) & (cdate.month <= 9):
        return 3
    elif (cdate.month >=10) & (cdate.month <= 12):
        return 4
    
banks['cyear'] = banks['Closing Date'].dt.year
banks['quar'] = banks['Closing Date'].apply(div_quar)
banks
