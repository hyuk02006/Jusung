def func_foreign(x):
    result = x[1] / x[0] * 100
    return result

def func_old(x):
    result = x[1] / x[0] * 100
    return result

popsSeoul_fil['외국인비율'] = popsSeoul_fil[['인구수','외국인']].apply(func_foreign,axis=1)
popsSeoul_fil['고령자비율'] = popsSeoul_fil[['인구수','고령자']].apply(func_old,axis=1)

popSeoul = popsSeoul_fil
